
int led = D7; 

void setup() 
{
    
  pinMode(led, OUTPUT);
  
}

void loop() 
{
    if(Particle.publish("Deakin_RIOT_SIT210_Photon_Buddy", "pat")==true)
    {
        for(int i=0;i<2;i++)
        {
            digitalWrite(led, HIGH);
            delay(500);
            digitalWrite(led, LOW);
            delay(500);
        }
    }

    delay(10000);
}
